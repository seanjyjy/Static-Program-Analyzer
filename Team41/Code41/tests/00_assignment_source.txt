procedure Peter {
    print present;
    future = 8 * present;
    call Stephen;
    y = present - 4;
    z = future + 2;
    while (present < 0) {
        x = z - present;
        if (present > x) then {
            present = z - 8 - (x * y) * z;
          x = x - 1; }
          else {
          y = present * y * z;
          call Stephen; }}
    future = x * y * z * present * future;
    print future; }

procedure Stephen {
    if (future > y) then {
	        present = z - 8 * present; }
       else {
	        future = 4 + present * future; }
     call Wong; }

procedure Wong {
    read x;
    read y; }
